SQL命令

命令 解释
create database day1; 创建一个名为“day1”的数据库
use day1; 进去day1数据库
create table test(
id varchar(20)  primary key ,
name varchar(30),
password varchar(30)
); 创建一个名为“test”的数据表
insert into test(id,name,password) values(‘001’,’lisi’,’123456’); 向test表插入一行数据
select * from test; 查看创建的test表的数据
update test set password=’xxxx’ where id=’001’; 将id为001的密码改为xxxx
drop table test; 删除test表
drop database day1; 删除day1数据库
